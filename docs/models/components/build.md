# Build

A build represents a game server artifact and its associated metadata.

## Example Usage

```typescript
import { Build } from "@hathora/cloud-sdk-typescript/models/components";

let value: Build = {
  buildTag: "0.1.14-14c793",
  regionalContainerTags: [
    {
      containerTag: "<value>",
      region: "Tokyo",
    },
  ],
  imageSize: 464108,
  status: "running",
  deletedAt: new Date("2023-01-16T03:54:06.760Z"),
  finishedAt: new Date("2024-09-16T20:14:13.716Z"),
  startedAt: new Date("2024-12-07T19:27:24.859Z"),
  createdAt: new Date("2024-09-24T20:17:00.614Z"),
  createdBy: "noreply@hathora.dev",
  buildId: 1,
  appId: "app-af469a92-5b45-4565-b3c4-b79878de67d2",
};
```
